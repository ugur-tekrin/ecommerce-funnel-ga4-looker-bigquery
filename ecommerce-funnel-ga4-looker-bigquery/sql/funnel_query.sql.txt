
-- GA4 E-commerce funnel dataset
-- session level aggregation
-- built for Looker Studio dashboard

WITH sessions_info AS (

SELECT
  user_pseudo_id,

  CONCAT(
    user_pseudo_id,
    CAST((
      SELECT value.int_value
      FROM UNNEST(event_params)
      WHERE key = 'ga_session_id'
    ) AS STRING)
  ) AS user_session_id,

  TIMESTAMP_MICROS(event_timestamp) AS session_start_time,
  DATE(TIMESTAMP_MICROS(event_timestamp)) AS session_date,
  FORMAT_DATE('%Y-%m', DATE(TIMESTAMP_MICROS(event_timestamp))) AS session_month,

  device.category AS device_category,
  device.language AS device_language,
  device.operating_system AS operating_system,

  geo.country AS country,

  traffic_source.source AS source,
  traffic_source.medium AS medium,
  traffic_source.name AS campaign,

  CONCAT(
    COALESCE(traffic_source.source, '(not set)'),
    ' / ',
    COALESCE(traffic_source.medium, '(not set)')
  ) AS source_medium,

  (
    SELECT value.string_value
    FROM UNNEST(event_params)
    WHERE key = 'page_location'
  ) AS landing_page

FROM `bigquery-public-data.ga4_obfuscated_sample_ecommerce.events_*`
WHERE event_name = 'session_start'

),

events AS (

SELECT
  CONCAT(
    user_pseudo_id,
    CAST((
      SELECT value.int_value
      FROM UNNEST(event_params)
      WHERE key = 'ga_session_id'
    ) AS STRING)
  ) AS user_session_id,

  event_name

FROM `bigquery-public-data.ga4_obfuscated_sample_ecommerce.events_*`
WHERE event_name IN (
  'session_start',
  'view_item',
  'add_to_cart',
  'begin_checkout',
  'add_shipping_info',
  'add_payment_info',
  'purchase'
)

),

session_level AS (

SELECT
  s.user_session_id,
  s.session_start_time,
  s.session_date,
  s.session_month,
  s.device_category,
  s.device_language,
  s.operating_system,
  s.country,
  s.source,
  s.medium,
  s.campaign,
  s.source_medium,
  s.landing_page,
  REGEXP_EXTRACT(s.landing_page, r'/([^/?#]+)(?:[/?#]|$)') AS landing_page_short,

  MAX(CASE WHEN e.event_name = 'session_start' THEN 1 ELSE 0 END) AS reached_session_start,
  MAX(CASE WHEN e.event_name = 'view_item' THEN 1 ELSE 0 END) AS reached_view_item,
  MAX(CASE WHEN e.event_name = 'add_to_cart' THEN 1 ELSE 0 END) AS reached_add_to_cart,
  MAX(CASE WHEN e.event_name = 'begin_checkout' THEN 1 ELSE 0 END) AS reached_begin_checkout,
  MAX(CASE WHEN e.event_name = 'add_shipping_info' THEN 1 ELSE 0 END) AS reached_add_shipping_info,
  MAX(CASE WHEN e.event_name = 'add_payment_info' THEN 1 ELSE 0 END) AS reached_add_payment_info,
  MAX(CASE WHEN e.event_name = 'purchase' THEN 1 ELSE 0 END) AS reached_purchase

FROM sessions_info s
LEFT JOIN events e
  ON s.user_session_id = e.user_session_id

GROUP BY
  s.user_session_id,
  s.session_start_time,
  s.session_date,
  s.session_month,
  s.device_category,
  s.device_language,
  s.operating_system,
  s.country,
  s.source,
  s.medium,
  s.campaign,
  s.source_medium,
  s.landing_page

),

funnel_long AS (

SELECT
  *,
  'session_start' AS funnel_step,
  1 AS funnel_order,
  reached_session_start AS step_reached
FROM session_level

UNION ALL

SELECT
  *,
  'view_item' AS funnel_step,
  2 AS funnel_order,
  reached_view_item AS step_reached
FROM session_level

UNION ALL

SELECT
  *,
  'add_to_cart' AS funnel_step,
  3 AS funnel_order,
  reached_add_to_cart AS step_reached
FROM session_level

UNION ALL

SELECT
  *,
  'begin_checkout' AS funnel_step,
  4 AS funnel_order,
  reached_begin_checkout AS step_reached
FROM session_level

UNION ALL

SELECT
  *,
  'add_shipping_info' AS funnel_step,
  5 AS funnel_order,
  reached_add_shipping_info AS step_reached
FROM session_level

UNION ALL

SELECT
  *,
  'add_payment_info' AS funnel_step,
  6 AS funnel_order,
  reached_add_payment_info AS step_reached
FROM session_level

UNION ALL

SELECT
  *,
  'purchase' AS funnel_step,
  7 AS funnel_order,
  reached_purchase AS step_reached
FROM session_level

)

SELECT
  *,
  reached_purchase AS purchase_flag,
  (
    reached_session_start +
    reached_view_item +
    reached_add_to_cart +
    reached_begin_checkout +
    reached_add_shipping_info +
    reached_add_payment_info +
    reached_purchase
  ) AS funnel_completed_steps
FROM funnel_long
WHERE step_reached = 1
